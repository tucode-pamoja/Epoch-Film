## Part 4: Database Schema Additions

```sql
-- Quests
CREATE TABLE quests (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  type VARCHAR(20) NOT NULL, -- DAILY, WEEKLY, SEASONAL, SPECIAL
  title VARCHAR(200) NOT NULL,
  description TEXT,
  xp_reward INTEGER DEFAULT 0,
  badge_reward VARCHAR(50),
  requirements JSONB,
  starts_at TIMESTAMPTZ,
  expires_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- User Quest Progress
CREATE TABLE user_quests (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  quest_id UUID REFERENCES quests(id),
  status VARCHAR(20) DEFAULT 'ACTIVE', -- ACTIVE, COMPLETED, EXPIRED
  progress JSONB,
  completed_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- User XP & Level
ALTER TABLE users ADD COLUMN xp INTEGER DEFAULT 0;
ALTER TABLE users ADD COLUMN level INTEGER DEFAULT 1;

-- Memory Reels
CREATE TABLE memory_reels (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  bucket_id UUID REFERENCES buckets(id),
  title VARCHAR(200),
  music_url VARCHAR(500),
  settings JSONB, -- transition, duration, etc.
  video_url VARCHAR(500), -- generated video
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Inspiration Feed (public dreams)
CREATE TABLE public_dreams (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title VARCHAR(200) NOT NULL,
  category VARCHAR(50),
  description TEXT,
  image_url VARCHAR(500),
  participant_count INTEGER DEFAULT 0,
  completion_count INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- User follows public dream
CREATE TABLE dream_participants (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  dream_id UUID REFERENCES public_dreams(id),
  bucket_id UUID REFERENCES buckets(id), -- linked personal bucket
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, dream_id)
);
```
